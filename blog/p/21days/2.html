结论
答案为：（ans为最短循环子串的长度） 

ans = (n-nxt[n])%n ? n : n-nxt[n];
代码
注意此题中要求的是最短循环节的出现次数 

#define MAXN 1000010

int n;
string s;

int nxt[MAXN];
void Next(){
	nxt[1] = 0;
	Forx(i,2,n){
		int prelen = nxt[i-1];
		while( prelen && s[prelen+1] != s[i] )
			prelen = nxt[prelen];
		nxt[i] = prelen + (s[prelen+1]==s[i]);
	}
}

int main(){
	FILEIO();

	while(1){
		In.Readstring(s);
		if( s == "." ) END();
		n = s.length();
		s = " "+s;
		Next();
		//Print(nxt,1,n);
		int ans = n - nxt[n];
		if(n%ans){ printf("%d\n",1); }
		else{ printf("%d\n",n/ans); }
	}

	END();
}
